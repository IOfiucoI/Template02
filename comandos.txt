git init - Se inializa el repositorio
git status - Verifica cambios
git config --global user.name "Marco"
git config --global user.email "marco_rg@msn.com"
git commit -m "Inicio de repositorio"
git diff DETECTA DIFERENCIAS
git log LISTA DE VERSIONES
git checkout --. RECUPERA ARCHIVOS EN COMMIT
git checkout --index.html ARCHIVO ESPECIFICO
git remote add origin URL DE REPOSITORIO
git pull Actualiza el repositorio si hubo cambnios (Download)
git push -u origin master ACTUALIZA REPOSITORIO (UPDATE) 
git clone (URL para copiar repositorio o por ZIP)
